```python
# Hello.py
class fighter: # 캐릭터 생성

    def stat(self, pow, int, spd):
        self.pow = pow
        self.int = int
        self.spd = spd

    def showstat(self):
        print('파워: ', self.pow)
        print('지능: ', self.int)
        print('민첩: ', self.spd)

    def __init__(self, pow, int, spd): # 객체 생성시 실행되는 함수
        self.stat(pow, int, spd)
        print('당신의 캐릭터가 생성되었습니다')

    def __del__(self):
        print('당신의 캐릭터가 제거되었습니다') # 객체가 사라질 때 나오는 함수
    
    def __repr__(self):
        return '캐릭터가 당신에게 인사합니다' # 객체가 이를 리턴한다

```

```python
# create.py
import Hello

fighter = Hello.fighter(30, 20, 15) # 당신의 캐릭터가 생성되었습니다

print(fighter.showstat())
''' 
파워:  30
지능:  20
민첩:  15 
'''

print(fighter) # 캐릭터가 당신에게 인사합니다

del fighter # 당신의 캐릭터가 제거되었습니다
 
```
