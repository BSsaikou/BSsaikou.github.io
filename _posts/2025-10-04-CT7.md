```python
f = open('C:\\python\\test.txt')  # hello world 라는 내용을 가진 test.txt 파일을 만들어놨다
print(f.read())

print('hello\nworld') # 줄바꿈

```

```python
f = open('C:\\python\\test.txt', 'w') # 쓰기모드(전에 있던 내용은 지워진다)

f.write('thank you') 

f.close()

f = open('C:\\python\\test.txt', 'r') # 읽기모드

test = f.read()

print(test) # thank you

f = open('C:\\python\\test.txt', 'a') # 쓰기모드(전에 있던 내용은 그대로 두고 추가 쓰기)

f.write('\nsee you next time')
f.close()

```

readline()

```python
f = open('C:\\python\\test.txt', 'r')

line = f.readline()  # \n 이 나올때까지만 읽는다
sec_line = f.readline()
f.close()

print(line)  # thank you
print(sec_line) # see you next time

```

readlines() 

```python
f = open('C:\\python\\test.txt', 'r') 

line_list = f.readlines() # \n 이 나올때마다 리스트에 추가한다

print(line_list) # ['thank you\n', 'see you next time']


```
