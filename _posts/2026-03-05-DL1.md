## AND, NAND, OR 게이트 구현
```python
import numpy as np

# 가중치와 임계치를 임의로 넣어서 게이트를 구현
def AND(x1, x2):
    x = np.array([x1, x2])
    w = np.array([0.5, 0.5])
    b = -0.7
    if np.sum(x*w) + b > 0:
        return 1
    else:
        return 0

def NAND(x1, x2):
    x = np.array([x1, x2])
    w = np.array([-0.5, -0.5])
    b = 0.7
    if np.sum(x*w) + b > 0:
        return 1
    else:
        return 0

def OR(x1, x2):
    x = np.array([x1, x2])
    w = np.array([1, 1])
    b = -0.9
    if np.sum(x*w) + b > 0:
        return 1
    else:
        return 0
```

```python
# 검증 및 보기 편하게 함수화
def AND_print():
    AND_arr = np.array([AND(0,0), AND(1,0), AND(0,1), AND(1,1)])
    print("this is And gate:" , AND_arr)

def NAND_print():
    NAND_arr = np.array([NAND(0,0), NAND(1,0), NAND(0,1), NAND(1,1)])
    print("this is Nand gate:" , NAND_arr)

def OR_print():
    OR_arr = np.array([OR(0,0), OR(1,0), OR(0,1), OR(1,1)])
    print("this is OR gate:" , OR_arr)

AND_print()
NAND_print()
OR_print()
```

XOR 게이트는 그냥 퍼셉트론으로는 구현이 불가능하다.

아무리 임계치랑 가중치를 설정하려고 해도 퍼셉트론은 선형 영역(직선)만을 표현할 수 있기에 XOR 게이트는 구현할 수 없다.

이를 구현하기 위해서는 다층 퍼셉트론이여야 한다.
